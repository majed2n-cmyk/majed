/* ========= فك قفل الصوت بأول تفاعل (غير ملحوظ) ========= */
function unlockAudioOnce() {
    if (sessionStorage.getItem("audioUnlocked") === "yes") return;

    const audio = document.createElement("audio");
    audio.src = "data:audio/mp3;base64,//uQxAAAAAAAAAAAAAAAAAAAAAAAWGluZwAAAA8AAAACAAACcQCA...";
    audio.volume = 0.01;

    audio.play().then(() => {
        sessionStorage.setItem("audioUnlocked", "yes");
        console.log("🔓 audio unlocked");
    }).catch(()=>{});
}

/* التنقل مع فك الصوت */
function go(page){
    unlockAudioOnce();
    setTimeout(() => {
        window.location.href = page;
    }, 80);
}
